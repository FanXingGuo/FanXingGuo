    // 使用cout输出

int main(int argc,char ** argv) {
    //使用cin输入,基本类型、字符数组、string类型

    int num=0;
    char str[20]={0};
    cout<<"Input a num:";
    cin>>num;
    cout<<"Input a str:";
    cin>>str;
    cout<<str<<endl;
    cout<<num<<endl;
    return 0;
}
