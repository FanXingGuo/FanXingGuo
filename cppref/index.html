#include <iostream>
#include <string>
#include <stdio.h>
#include "../lib/Student.h"
using namespace std;

void func(Object obj){// 值传递，这里调用了拷贝构造函数
    cout<<obj.getNum()<<endl;
}
void func0(Object &obj){ // 这里使用的是引用，没有调用拷贝构造函数
    cout<<obj.getNum()<<endl;
}

int main(int argc,char ** argv) {
    Object obj(7);
    func(obj);
    func0(obj);
    return 0;
}
//输出：
///Object::Object(const Object &obj)
//7
//7

拷贝构造函数

拷贝构造函数是一种特殊的构造函数，其唯一的形参必须是构造对象的引用。

编译器会自动生成拷贝构造函数，会把参数所有非静态成员传递给对象。

被自动调用的情况：

对象作为参数，以值传递方式传入函数体。

对象作为函数返回值，以值传递方式从函数返回。

对象用于给另一对象初始化。

#include "Student.h"
#include <iostream>
#include <string.h>

using namespace std;

Object::Object(int num) :num(num){}
int Object::getNum() {
    return num;
}
void Object::setNum(int num) {
    this->num=num;
}
Object::Object(const Object &obj) {
    cout<<"Object::Object(const Object &obj)"<<endl;
    num=obj.num; // 加上这一句，可以使参数传递，打印结果 0 变 7
}
